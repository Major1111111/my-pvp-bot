Хорошо, вот только JavaScript код, без HTML и CSS.

Этот код предполагает, что HTML-структура с элементами (#timer, #wheel, #player-list, #make-bet-btn, #winner-overlay и т.д.) и CSS стили уже присутствуют на вашей странице.

// Инициализация Socket.IO
const socket = io();
// Инициализация Telegram Web App
const tg = window.Telegram.WebApp;
tg.expand();

// Задаем более надежные дефолты для user на случай, если tg.initDataUnsafe.user недоступен
const user = tg.initDataUnsafe?.user || { 
    username: "User" + Math.floor(Math.random() * 10000), // Генерируем уникальный юзернейм для теста
    photo_url: "https://via.placeholder.com/100", // Заглушка для фото
    id: Date.now() // Уникальный ID
};

// Получаем элементы Canvas и его контекст для рисования рулетки
const canvas = document.getElementById('wheel');
const ctx = canvas.getContext('2d');

let players = []; // Массив игроков в текущей игре
let images = {}; // Объект для кэширования загруженных изображений игроков
let currentGameId = 0; // ID текущей игры

// Обработка события обновления таймера
socket.on('timerUpdate', (data) => {
    document.getElementById('timer').innerText = data.timeLeft;
    currentGameId = data.gameId;
});

// Обработка события инициализации (первичная загрузка данных игры)
socket.on('init', (data) => {
    console.log('Init data:', data);
    players = data.players;
    updateUI(); // Обновляем интерфейс после получения данных
});

// Обработка события обновления списка игроков
socket.on('updatePlayers', (p) => {
    console.log('Players updated:', p);
    players = p;
    updateUI(); // Обновляем интерфейс
});

// Обработка события начала вращения рулетки
socket.on('startSpin', (data) => {
    console.log('Received startSpin event with data:', data); 
    // Валидация полученных данных для предотвращения ошибок
    if (!data || typeof data.seed === 'undefined' || !data.winner || typeof data.bank === 'undefined') {
        console.error('startSpin event received with invalid data:', data);
        // Можно добавить alert или другое уведомление пользователю/админу
        return;
    }

    // Отключаем кнопку ставки и делаем ее полупрозрачной
    document.getElementById('make-bet-btn').disabled = true;
    document.getElementById('make-bet-btn').style.opacity = "0.5";
    // Запускаем анимацию рулетки
    animateWheel(data.seed, data.winner, data.bank);
});

// Обработка события сброса игры
socket.on('resetGame', () => {
    console.log('Game reset.');
    players = []; // Очищаем список игроков
    // Включаем кнопку ставки и возвращаем ей полную прозрачность
    document.getElementById('make-bet-btn').disabled = false;
    document.getElementById('make-bet-btn').style.opacity = "1";
    updateUI(); // Обновляем интерфейс
});

// Функция обновления пользовательского интерфейса (списка игроков и рулетки)
function updateUI() {
    const list = document.getElementById('player-list');
    const total = players.reduce((s, p) => s + p.amount, 0); // Общая сумма ставок

    // Формируем HTML для списка игроков
    list.innerHTML = players.map(p => `
        <div class="player-item">
            <div class="item-top">
                <div class="item-user">
                    <img src="${p.photo}">
                    <div class="names">
                        <span class="nick">@${p.username}</span>
                        <span class="role">${p.username === user.username ? 'Ты' : 'Игрок'}</span>
                    </div>
                </div>
                <div class="item-stats">
                    <div class="perc">${total > 0 ? ((p.amount/total)*100).toFixed(1) : 0}%</div>
                    <div class="sum">${p.amount.toFixed(1)} TON</div>
                </div>
            </div>
            <div class="item-bet-badge"><span>💎</span> ${p.amount.toFixed(1)}</div>
        </div>
    `).join('');

    // Загружаем изображения игроков для рулетки
    players.forEach(p => {
        if(!images[p.photo]) {
            const img = new Image(); 
            img.src = p.photo;
            img.onload = () => drawWheel(0); // Перерисовать колесо после загрузки изображения
            img.onerror = () => {
                console.warn(`Failed to load image for ${p.username}: ${p.photo}`);
                // Используем заглушку, если изображение не загрузилось
                images[p.photo] = new Image();
                images[p.photo].src = "https://via.placeholder.com/100"; 
                drawWheel(0);
            };
            images[p.photo] = img;
        }
    });
    drawWheel(0); // Рисуем колесо статично
}

// Функция отрисовки рулетки
function drawWheel(rotation) {
    const total = players.reduce((s, p) => s + p.amount, 0);
    const centerX = 250, centerY = 250, radius = 240; // Центр и радиус колеса
    ctx.clearRect(0,0,500,500); // Очищаем canvas перед каждой отрисовкой

    // Если игроков нет, рисуем пустой круг
    if(players.length === 0) {
        ctx.beginPath(); ctx.fillStyle = '#1c1c1e';
        ctx.arc(centerX, centerY, radius, 0, Math.PI*2); ctx.fill();
        return;
    }

    let startAngle = rotation - Math.PI/2; // Начальный угол для первого сегмента
    players.forEach(p => {
        const slice = (p.amount / total) * Math.PI * 2; // Размер сегмента
        ctx.beginPath(); 
        ctx.fillStyle = p.color || '#cccccc'; // Цвет сегмента, по умолчанию серый
        ctx.moveTo(centerX, centerY);
        ctx.arc(centerX, centerY, radius, startAngle, startAngle + slice);
        ctx.fill();

        // Рисуем рамку вокруг сегмента
        ctx.strokeStyle = '#2c2c2e'; 
        ctx.lineWidth = 2; 
        ctx.stroke();

        // Рисуем аватар игрока в центре сегмента
        if (images[p.photo] && images[p.photo].complete && images[p.photo].naturalHeight !== 0) {
            ctx.save(); 
            ctx.translate(centerX, centerY);
            ctx.rotate(startAngle + slice/2);
            ctx.beginPath(); 
            ctx.arc(radius*0.75, 0, 25, 0, Math.PI*2); 
            ctx.closePath();
            ctx.clip(); // Обрезаем изображение по кругу
            ctx.drawImage(images[p.photo], radius*0.75-25, -25, 50, 50);
            ctx.restore();
        }
        startAngle += slice; // Переходим к следующему сегменту
    });
}

// Функция анимации рулетки
function animateWheel(seed, winner, bank) {
    console.log('Starting wheel animation with seed:', seed, 'winner:', winner.username);
    let start = null;
    const duration = 6000; // Длительность анимации (6 секунд)
    // Общее количество оборотов, используем seed для небольшого разнообразия
    const totalRotation = (20 + (seed % 10)) * Math.PI * 2; 
    let currentRotation = 0;

    function step(timestamp) {
        if (!start) start = timestamp;
        const progress = Math.min((timestamp - start) / duration, 1);
        const ease = 1 - Math.pow(1 - progress, 3); // Функция замедления (easing function)
        currentRotation = ease * totalRotation; // Расчет текущего угла поворота
        drawWheel(currentRotation); // Перерисовываем колесо

        if (progress < 1) {
            requestAnimationFrame(step); // Продолжаем анимацию
        } else {
            console.log('Animation finished. Final rotation:', currentRotation);
            // Показываем окно победителя
            showWinnerUI(winner, bank);
        }
    }
    requestAnimationFrame(step); // Запускаем анимацию
}

// Функция отображения интерфейса победителя
function showWinnerUI(winner, bank) {
    const total = players.reduce((s, p) => s + p.amount, 0);
    document.getElementById('win-game-id').innerText = currentGameId;
    document.getElementById('win-u-img').src = winner.photo;
    document.getElementById('win-u-nick').innerText = `@${winner.username}`;
    document.getElementById('win-u-sum').innerText = bank.toFixed(2);
    document.getElementById('win-card-sum').innerText = bank.toFixed(2);
    document.getElementById('win-u-perc').innerText = ((winner.amount / total) * 100).toFixed(1) + '%';

    const title = document.querySelector('.win-title');
    if(winner.username === user.username) {
        title.innerHTML = "Поздравляем! Ты<br>победил!";
        document.querySelector('.u-you').style.display = 'block';
    } else {
        title.innerHTML = "Игра окончена!<br>Победитель:";
        document.querySelector('.u-you').style.display = 'none';
    }
    document.getElementById('winner-overlay').style.display = 'flex'; // Показываем оверлей
}

// Функции для управления модалками и кнопкой "Готово"
function closeWinner() { document.getElementById('winner-overlay').style.display = 'none'; }
function openModal() { document.getElementById('modal').style.display = 'flex'; }
function closeModal() { document.getElementById('modal').style.display = 'none'; }

// Функция подтверждения ставки
function confirmBet() {
    const amountInput = document.getElementById('bet-amount');
    const amount = parseFloat(amountInput.value);

    if (isNaN(amount) || amount < 0.1) {
        alert("Минимальная ставка 0.1 TON. Пожалуйста, введите корректное число.");
        return;
    }

    // Отправляем событие 'makeBet' на сервер
    socket.emit('makeBet', {
        username: user.username,
        photo: user.photo_url || "https://via.placeholder.com/100",
        amount: amount,
        // Генерируем случайный цвет для игрока, если сервер его не предоставляет
        color: '#' + Math.floor(Math.random()*16777215).toString(16) 
    });
    closeModal(); // Закрываем модалку
    amountInput.value = '0.1'; // Сбрасываем значение поля ввода
}
